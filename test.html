MySQL Database

MySQL is a relational database management system used to store structured data.

Create Database

Create a new database.

CREATE DATABASE db_name;
CREATE DATABASE school;

Drop Database

Deletes a database permanently.

DROP DATABASE db_name;
DROP DATABASE school;

Create Table

Create a table inside a database.

CREATE TABLE table_name (
  column1 datatype,
  column2 datatype
);

CREATE TABLE students (
 id INT PRIMARY KEY,
 name VARCHAR(50),
 age INT
);

Drop Table

Delete a table.

DROP TABLE table_name;
DROP TABLE students;

Alter Table

Modify table structure.

ALTER TABLE table_name ADD column datatype;
ALTER TABLE students ADD email VARCHAR(100);

SELECT

Retrieve data from a table.

SELECT column FROM table;
SELECT * FROM students;

WHERE

Filter records.

SELECT * FROM table WHERE condition;
SELECT * FROM students WHERE age = 20;

AND OR NOT

Combine conditions.

SELECT * FROM students 
WHERE age = 20 AND name = 'Ali';

ORDER BY

Sort results.

SELECT * FROM students ORDER BY name ASC;

INSERT INTO

Insert data.

INSERT INTO students (id,name,age)
VALUES (1,'Ali',22);

NULL Values

Check missing values.

SELECT * FROM students WHERE email IS NULL;

UPDATE

Modify existing records.

UPDATE students 
SET age = 23 
WHERE id = 1;

DELETE

Delete records.

DELETE FROM students WHERE id = 1;

LIMIT

Limit returned rows.

SELECT * FROM students LIMIT 5;

MIN and MAX

SELECT MIN(age) FROM students;
SELECT MAX(age) FROM students;

COUNT, AVG, SUM

SELECT COUNT(*) FROM students;
SELECT AVG(age) FROM students;
SELECT SUM(age) FROM students;

LIKE

Search pattern.

SELECT * FROM students WHERE name LIKE 'A%';

Wildcards

% represents multiple characters, _ represents one character.

SELECT * FROM students WHERE name LIKE '_li';

IN

Match multiple values.

SELECT * FROM students WHERE age IN (20,21,22);

BETWEEN

Filter range.

SELECT * FROM students WHERE age BETWEEN 18 AND 25;
